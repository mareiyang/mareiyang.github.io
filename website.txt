            <h4 class="font-bold text-white mb-2">Abstract</h4>
            <p class="text-gray-300 italic">
              I examine the effect of H-1B visa lottery outcomes on firm concentration and business diversity outcomes at the county level in the United States, leveraging a natural experiment. I combine H-1B data from the Department of Labor and US Citizenship and Immigration Services with data from the US Census Bureau's County Business Patterns to analyze changes in firm concentration and business diversity. I measure business diversity using two indices: the Herfindahl-Hirschman Index and the Shannon Diversity Index. Using a continuous difference-indifferences design, I find that counties with a higher lottery win rate experienced a 6.4% decline in the Herfindahl-Hirschman Index and a 2.7% increase in the Shannon Diversity Index. These findings suggest that high skilled immigration contributes to economic diversification, supporting the broader argument that the H-1B visa plays a role beyond filling labor shortages.

            </p>

  <script>
    // Page switching + nav active state + aria-current
    function showPage(pageId) {
      // toggle pages
      document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));
      document.getElementById(pageId).classList.add('active');

      // toggle nav active state
      document.querySelectorAll('.nav-link').forEach(link => {
        link.classList.remove('active');
        link.removeAttribute('aria-current');
      });
      document.querySelectorAll(`a[href="#${pageId}"]`).forEach(link => {
        link.classList.add('active');
        link.setAttribute('aria-current', 'page');
      });
    }

    // on load, show home
    window.onload = function() {
      showPage('home');
    };

    // Mobile menu
    const mobileMenuButton = document.getElementById('mobile-menu-button');
    const mobileMenu = document.getElementById('mobile-menu');
    mobileMenuButton.addEventListener('click', () => mobileMenu.classList.toggle('hidden'));
    function closeMenu() { mobileMenu.classList.add('hidden'); }
    document.querySelectorAll('#mobile-menu a').forEach(link => {
      link.addEventListener('click', (e) => {
        const href = e.currentTarget.getAttribute('href');
        const pageId = href && href.startsWith('#') ? href.slice(1) : 'home';
        showPage(pageId);
        closeMenu();
      });
    });
  </script>
